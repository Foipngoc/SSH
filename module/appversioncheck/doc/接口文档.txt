【功能】发布应用新版本
【接口】http://ip/project/module/AppVersionCheck/publishAppVersion
【必选参数】
	appid //应用id
	autogenvcode //是否自动生成版本号 1/0
	versioncode //版本号
	versionname //版本名
	updatelog //更新日志
	updatetype //更新方式
	autoset //是否自动将该版本设置为最新 1/0
	downloadpath //APP下载到本地路径，通常为空，由APP自己选择
	autoinstall //下载完立即安装 1/0
【可选参数】
【输出】
	BaseResult(-1, "参数错误")
	BaseResult(3, "未找到应用")
	BaseResult(2,"APP上传失败")
	BaseResult(4, "文件保存失败")
	BaseResult(5, "MD5校验失败")
	BaseResult(1, "版本号必须大于当前发布版本中最大版本号)
	BaseResult(0, "发布成功")
